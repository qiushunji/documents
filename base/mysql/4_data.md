```sql
CREATE TABLE `students` (
    `id` int(11) NOT NULL,
    `sName` varchar(255) NOT NULL,
    `sCampus` varchar(255) DEFAULT NULL,
    `sCollege` varchar(255) DEFAULT NULL,
    `sMajor` varchar(255) DEFAULT NULL,
    `sClass` varchar(255) DEFAULT NULL,
    `sType` varchar(255) NOT NULL,
    `sStartDate` varchar(255) NOT NULL,
    `sEndDate` varchar(255) NOT NULL,
```

#### 导师双向选择表

```sql
CREATE TABLE `tutors` (
  `id` int(18) NOT NULL AUTO_INCREMENT,
  `iTeacherID` int(18) NOT NULL,
  `iStudentID` int(18) NOT NULL,
  `sStartDate` varchar(255) DEFAULT NULL,
  `sEndDate` varchar(255) DEFAULT NULL,
  PRIMARY KEY (`id`)
) ENGINE=InnoDB AUTO_INCREMENT=2018009002 DEFAULT CHARSET=utf8;
```

为学生分配导师

```sql
INSERT INTO `document`.`tutors` ( `iTeacherID`, `iStudentID`, `sStartDate`, `sEndDate` ) (
SELECT
	teach.id AS iTeacherID,
	stu.id AS iStudentID,
	stu.sStartDate,
	stu.sEndDate 
FROM
	`document`.`students` stu,
	`document`.`teacher` teach 
WHERE
	stu.sMajor = teach.sMajor 
	)
```
